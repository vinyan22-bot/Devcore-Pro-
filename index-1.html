// DevCore Pro AI Review SDK
import { DevCore } from '@devcore/sdk'
 
const dc = new DevCore({
  apiKey: process.env.DEVCORE_KEY,
  repo: 'github.com/acme/api',
  mode: 'ai-strict'
})
 
// Run full intelligence scan
const report = await dc.scan({
  targets: ['security', 'perf', 'logic'],
  diff: 'HEAD~1',
  autoFix: true
})
 
console.log(report.issues) // ← 0 criticals
console.log(report.riskScore) // ← 3/100
 
// Gate: deploy only if score < 20
if (report.riskScore < 20) {
  await dc.deploy({ env: 'prod' })
}
